<?xml version=”1.0” encoding=”UTF-8?><poem>	<item>      <title><b>The Tyger</b></title>      <author>BY WILLIAM BLAKE</author>      </item>      </lb>      <item>      <stanza1>      	<verse1>Tyger Tyger, burning bright,</verse1>      	<verse2>In the forests of the night;</verse2>      	<verse3>What immortal hand or eye,</verse3>      	<verse4>Could frame thy fearful symmetry</verse4>      </stanza1>      </lb>      <stanza2>           <verse1>In what distant deeps or skies.</verse1>      	<verse2>Burnt the fire of thine eyes?</verse2>      	<verse3>On what wings dare he aspire?</verse3>      	<verse4>What the hand, dare seize the fire?</verse4>      </stanza2>      </lb>      <stanza3>      <verse1>And what shoulder, & what art,</verse1>      	<verse2>Could twist the sinews of thy heart?</verse2>      	<verse3>And when the heart began to beat.</verse3>      	<verse4>What dread hand? & what dread feet?</verse4>      </stanza3>      </lb>      <stanza4>      <verse1>What the hammer? What the chain,</verse1>      	<verse2>In what furnance was thy brain?</verse2>      	<verse3>What the anvil? What dread grasp.</verse3>      	<verse4>Dare its deadly terrors clasp?</verse4>      </stanza4>      </lb>      <stanza5>      <verse1>When the stars threw down their spears</verse1>      	<verse2>And water’d heaven with their tears:</verse2>      	<verse3>Did he smile his work to see?</verse3>      	<verse4>Did he who made the Lamb make thee?</verse4>      </stanza5>      </lb>      <stanza6>      <verse1>Tyger Tyger burning bright,</verse1>      	<verse2>In the forests of the night:</verse2>      	<verse3>What immortal hand or eye,</verse3>      	<verse4>Dare frame thy fearful symmetry?</verse4>      </stanza6>      </item></poem>
